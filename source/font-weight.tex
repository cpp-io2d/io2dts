\rSec0 [\iotwod.font.weight] {Enum class \tcode{font_weight}}

\rSec1 [\iotwod.font.weight.summary] {\tcode{font_weight} Summary}

\pnum
The \tcode{font_weight} enum class specifies the weight requested for rendering 
text.
See Table~\ref{tab:\iotwod.font.weight.meanings} for the meaning of each
\tcode{} enumerator.

\rSec1 [\iotwod.font.weight.synopsis] {\tcode{font_weight} Synopsis}

\begin{codeblock}
namespace std { namespace experimental { namespace io2d { inline namespace v1 {
  enum class font_weight {
    normal,
    bold
  };
} } } } // namespaces std::experimental::io2d::v1
\end{codeblock}
