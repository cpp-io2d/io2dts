\rSec0 [\iotwod.text.fontdb] {Class template \tcode{basic_font_database}}

\rSec1 [\iotwod.text.fontdb.summary] {\tcode{basic_font_database} summary}

\pnum
\indexlibrary{\idxcode{basic_font_database}}%
The \tcode{basic_font_database} class provides a way to get a list of all font families available in the environment the program is running in.

\pnum
It has no public data.

\pnum
The data are stored in an object of type \tcode{typename GraphicsSurfaces::text::font_database_data_type}. It is accessible using the \tcode{data} member function.

\rSec1 [\iotwod.text.fontdb.matching]{\tcode{basic_font_database} matching}

\rSec1 [\iotwod.text.fontdb.synopsis] {\tcode{basic_font_database} synopsis}

\begin{codeblock}
namespace @\fullnamespace{}@ {
  template <class GraphicsSurfaces>
  class basic_font_database {
  public:
    using data_type = typename GraphicsSurfaces::text::font_database_data_type;

    // \ref{\iotwod.text.fontdb.ctor}, construct:
    basic_font_database();

    // \ref{\iotwod.text.fontdb.acc}, accessors:
    const data_type& data() const noexcept;
    data_type& data() noexcept;

    // \ref{\iotwod.text.fontdb.obs}, observers:
    vector<string> get_families() const noexcept;
  };
}
\end{codeblock}
