\rSec0 [\iotwod.arc] {Class \tcode{arc}}

\rSec1 [\iotwod.arc.general] {In general}

\pnum
\indexlibrary{\idxcode{arc}}%
The class \tcode{arc} describes a figure item that is a segment.

\pnum
It has a \term{radius} of type \tcode{point_2d}, a \term{rotation} of type \tcode{float}, and a \term{start angle} of type \tcode{float}.

\pnum
It forms a portion of the circumference of a circle. The centre of the circle is implied by the start point, the radius and the start angle of the arc.

\end{itemdecl}
\begin{itemdescr}
\pnum
\returns
As-if:
\begin{codeblock}
auto lmtx = m;
lmtx.m20 = 0.0f;
lmtx.m21 = 0.0f;
auto centerOffset = point_for_angle(two_pi<float> - _Start_angle, _Radius);
centerOffset.y(-centerOffset.y());
return cpt - centerOffset * lmtx;
\end{codeblock}
\end{itemdescr}

  const noexcept;
\end{itemdecl}
\begin{itemdescr}
\pnum
\returns
As-if:
\begin{codeblock}
auto lmtx = m;
auto tfrm = matrix_2d::init_rotate(_Start_angle + _Rotation);
lmtx.m20 = 0.0f;
lmtx.m21 = 0.0f;
auto pt = (_Radius * tfrm);
pt.y(-pt.y());
return cpt + pt * lmtx;
\end{codeblock}
\end{itemdescr}
